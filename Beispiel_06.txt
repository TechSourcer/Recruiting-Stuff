SELECT us.*,tuser.Score
FROM Users us
JOIN
((SELECT Answers.OwnerUserId AS UserId, SUM (Answers.Score) AS Score
  FROM Tags
  JOIN PostTags ON Tags.Id = PostTags.TagId
  JOIN Posts ON Posts.Id = PostTags.PostId
  JOIN Posts Answers ON Answers.ParentId = Posts.Id
  WHERE Tags.TagName
  IN ('javascript', 'html', 'css', 'frontend') 
  GROUP By Answers.ownerUserId
  ) UNION
 (SELECT u.Id as UserId, 0 AS Score
 FROM Users u
 WHERE lower(u.AboutMe) like '%javascript%'
 ))
 tuser ON tuser.UserId = us.Id
 WHERE
 lower(location) like '%wolfsburg%' or
 lower (location) like '%braunschweig%' or
 lower (location) like '%berlin%'
 ORDER BY tuser.Score DESC
